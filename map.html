map.on('contextmenu', function(e) { const { lat, lng } = e.latlng; fetchData(lat.toFixed(5), lng.toFixed(5), e.latlng); }); async function fetchData(lat, lng, latlng) { const popupHtml = `
GPS: ${lat}, ${lng}
🌅 日出日落讀取中...
🌊 潮汐預報讀取中...
`; const popup = L.popup().setLatLng(latlng).setContent(popupHtml).openOn(map); // 1. 抓取日出日落 try { const sunRes = await fetch(`https://api.sunrise-sunset.org/json?lat=${lat}&lng=${lng}&formatted=0`); const sunData = await sunRes.json(); const sunrise = new Date(sunData.results.sunrise).toLocaleTimeString('zh-TW', {hour: '2-digit', minute:'2-digit'}); const sunset = new Date(sunData.results.sunset).toLocaleTimeString('zh-TW', {hour: '2-digit', minute:'2-digit'}); document.getElementById('sun-box').innerHTML = `🌅 日出 ${sunrise} | 日落 ${sunset}`; } catch (e) { document.getElementById('sun-box').innerText = "❌ 日出讀取失敗"; } // 2. 抓取潮汐 (深度解析安平站點) try { const tideUrl = `https://opendata.cwa.gov.tw/api/v1/rest/datastore/F-A0021-001?Authorization=${API_KEY}`; const res = await fetch(tideUrl); const data = await res.json(); // 深度搜尋：找出臺南市的安平測站 const tainan = data.records.Location.find(loc => loc.LocationName === "臺南市"); const anping = tainan.Station.find(st => st.StationName === "安平") || tainan.Station[0]; const now = new Date(); const nextTide = anping.TideForecast.TimePeriods.find(t => new Date(t.DateTime) > now); if (nextTide) { const tTime = nextTide.DateTime.split('T')[1].substring(0, 5); document.getElementById('tide-box').innerHTML = `🌊 下次 ${nextTide.TideType}：${tTime} (${nextTide.TideHeight || '--'}cm)`; } else { document.getElementById('tide-box').innerText = "⚠️ 暫無預報資料"; } } catch (e) { console.error(e); document.getElementById('tide-box').innerText = "❌ 潮汐解析失敗，請確認 API Key 權限"; } }
